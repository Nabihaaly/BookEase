# Use the official .NET image as a build stage
FROM mcr.microsoft.com/dotnet/aspnet:9.0 AS base
WORKDIR /app
EXPOSE 80

FROM mcr.microsoft.com/dotnet/sdk:9.0 AS build
WORKDIR /src

# Copy the csproj properly (folder + file)
COPY ["AppointmentBookingSystem/AppointmentBookingSystem.csproj", "AppointmentBookingSystem/"]

RUN dotnet restore "AppointmentBookingSystem/AppointmentBookingSystem.csproj"

# Copy everything
COPY . . 

# Build the project
WORKDIR "/src/AppointmentBookingSystem"
RUN dotnet build "AppointmentBookingSystem.csproj" -c Release -o /app/build

# Publish stage
FROM build AS publish
RUN dotnet publish "AppointmentBookingSystem.csproj" -c Release -o /app/publish

# Final image
FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "AppointmentBookingSystem.dll"]
